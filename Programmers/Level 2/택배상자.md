from collections import deque
def solution(order):
    
    stack = []
    order = deque(order)
    q = deque([ i for i in range(1,len(order)+1) ])
    cnt = 0
    # íë‚˜ ìŠ¤íƒì—ì„œ ëº„ ë•Œ ê²°êµ­ì€ í•´ë‹¹ ìˆ«ìì—ì„œ ê°€ì¥ ê°€ê¹Œìš´ ìš”ì†Œë§Œ ëº„ìˆ˜ ìˆë‹¤.
    while q or stack:
        
        if q and q[0] == order[0]:
            q.popleft()
            order.popleft()
            cnt += 1
        elif stack and stack[-1] == order[0]:
            stack.pop()
            order.popleft()
            cnt += 1
        else:
            if stack and stack[-1] > order[0]:
                break
            stack.append(q.popleft())
            
    return cnt
