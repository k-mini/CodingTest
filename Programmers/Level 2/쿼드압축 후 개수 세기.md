def solution(arr): # ë‹¤ì‹œ í’€ê¸°
    
    n, now = len(arr), len(arr)
    checked = [[False] * n for _  in range(n)]
    counter = Counter()
    for a in arr:
        counter += Counter(a)
        
    while now >= 2:
        for i in range(0, n, now):
            for j in range(0, n, now):
                if not checked[i][j]:
                    k, v  = quard_check(i, j, arr, now, checked)
                    if v != 0 :
                        counter[k] -= v - 1
            
        now //= 2
    return [counter[0], counter[1]]
def quard_check(x, y, arr, now, checked):
    num = arr[x][y]
    
    for i in range(x, x + now):
        for j in range(y, y + now):
            if num != arr[i][j]:
                return 0, 0
    
    for i in range(x, x + now):
        for j in range(y, y + now):
            checked[i][j] = True
    return num, now * now

def solution(arr):
    answer = [0, 0]

    def check(size, x, y):
        if size == 1:
            answer[arr[y][x]] += 1
            return
        else:
            first = arr[y][x]

            for dy in range(size):
                for dx in range(size):
                    if first != arr[y + dy][x + dx]:
                        check(size // 2, x, y)
                        check(size // 2, x + size // 2, y)
                        check(size // 2, x, y + size // 2)
                        check(size // 2, x + size // 2, y + size // 2)
                        return
            answer[first] += 1
    check(len(arr),0,0)
    return answer
