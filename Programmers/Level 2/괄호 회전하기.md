def isright(s):
    type1 = 0
    type2 = 0
    type3 = 0
    for word in s:
        if word == '(':
            type1 += 1
        elif word == ')':
            type1 -= 1
        elif word == '{':
            type2 += 1
        elif word == '}':
            type2 -= 1
        elif word == '[':
            type3 += 1
        else:
            type3 -= 1

            return False
    return True if type1 == 0 and type2 == 0 and type3 == 0 else False
