def solution(numbers):
    stack = []
    answer = [-1] * len(numbers)
    for i in range(len(numbers)):
        while stack and  numbers[i] > stack[-1][1]:
            idx , _ = stack.pop()
            answer[idx] = numbers[i]
        stack.append([i,numbers[i]])
    return answer

from heapq import *
def solution(numbers):
    
    n = len(numbers)
    answer = [-1] * n
    
    h = []
    
    for i in range(n):
        value = numbers[i]
        
        while h and value > h[0][0]:
            _, idx = heappop(h)
            answer[idx] = value
            
        heappush(h,[value, i])
    return answer
