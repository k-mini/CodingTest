from collections import Counter
def solution(n, works): # ë‹¤ì‹œí’€ê¸°

    if n >= sum(works):
        return 0
    works = Counter(works)
    
    for i in range(n):
        
        max_work = max(works.keys())
        
        if max_work != 0:
            works[max_work] -= 1
            works[max_work - 1 ] += 1
            if works[max_work] == 0:
                del works[max_work]
        else:
            return 0
    return sum([ w**2 * works[w] for w in works ])
