def convert_time(time):
    h,m,s = time[:2],time[3:5],time[6:8]
    return ((int(h)*60) + int(m)) * 60 + int(s)

def convert_str_time(t):
    h,t = divmod(t,3600)
    m,t = divmod(t,60)
    s = t
    return str(h).rjust(2,'0') + ':' + str(m).rjust(2,'0') + ':' + str(s).rjust(2,'0')
