import copy
def solution(key, lock):
    result = False
    for _ in range(4):
        key = rotate(key)
        result = test(key,lock)
        if result:
            break
    return result
def rotate(key):
    
    m = len(key)
    tmp = [ [ 0 for j in range(m) ] for i in range(m) ]
    
    for row in range(m):
        for col in range(m):
            tmp[col][m-1-row] = key[row][col]
    return tmp
def test(key,lock):
    m = len(key)
    n = len(lock)
    arr = [ [ 0 ] * (n+2*m-2) for i in range(n+2*m-2)]
    for i in range(n):
        for j in range(n):
            arr[m-1+i][m-1+j] = lock[i][j]
            
    for i in range(m-1+n):
        for j in range(m-1+n):
            tmp = copy.deepcopy(arr)
            for row in range(m):
                for col in range(m):
                    tmp[i+row][j+col] += key[row][col]
                    
            flag = False
            for r in range(n):
                for c in range(n):

                return True
             
    return False
