    tickets.sort( )
    path = []
    stack = []
    n = len(tickets)
    #visited = [ False for _ in range(n) ]
    
    for i in range(n ):
        depart,arrive = tickets[i]
        if depart == "ICN":
            
            visited = [ False for _ in range(n) ]
            visited[i] = True
            #print(tickets,i)
            stack.append(["ICN"] + [arrive] )
    
            while stack:
                now = stack[-1][-1]

                    if now == tt[0] and not visited[j]:
