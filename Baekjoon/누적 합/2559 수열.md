import sys
input = sys.stdin.readline
n,k = map(int,input().split())
arr = list(map(int,input().rstrip().split()))
answer = []
for i in range(1,n):
    arr[i] += arr[i-1]
    
answer.append(arr[k-1])

for j in range(k,n):
    answer.append(arr[j] - arr[j-k])
    
print(max(answer))
