import sys
from heapq import *
input = sys.stdin.readline
n,k = map(int,input().split())
INF = 1e10
graph,result = [],INF
visited = [False] * n
visited[k] = True
for _ in range(n):
    graph.append(list(map(int,input().split())))


# í”Œë¡œì´ë“œ ì›Œì…œ ì•Œê³ ë¦¬ì¦˜
for l in range(n):
    for i in range(n):
        for j in range(n):
            graph[i][j] = min(graph[i][l] + graph[l][j]   , graph[i][j])
        
def find_min(curr,count,cost):
    global result
    #print(curr,count,cost)
    if count == n:
        result = min(result,cost)
        return
    for i in range(n):
        if not visited[i]:
            visited[i] = True
            find_min(i, count+1, cost + graph[curr][i])
            visited[i] = False
#print(k)
find_min(k,1,0)
print(result)
