```
import sys
from collections import deque
input = sys.stdin.readline
n,m = map(int,input().rstrip().split())
visited = [ False for _ in range(101)]
ladder,snake = {},{}
q = deque()
for _ in range(n):
    a,b = map(int,input().rstrip().split())
    ladder[a] =b
for _ in range(m):
    a,b = map(int,input().rstrip().split())
    snake[a] =b
    
visited[1] = True
q.append((1,0))

while q:
    now, dist= q.popleft()
    
    if now == 100:
        print(dist)
        break
        
    for move in range(1,7):
        nxt = now + move
        if nxt > 100: break
        
        if not visited[nxt]: 
            visited[nxt] = True
            if nxt in ladder:
                visited[ladder[nxt]] = True
                nxt = ladder[nxt]
            if nxt in snake:
                visited[snake[nxt]] = True
                nxt = snake[nxt]
            q.append((nxt,dist+1))
```
