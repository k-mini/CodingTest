import sys
input = sys.stdin.readline

def solution(n,arr):
    
    dp = [0]* n
    if len(arr) == 1:
        print(arr[0])
    elif len(arr) == 2:
        print(arr[0]+arr[1])
    else:
        dp[0] = arr[0]
        dp[1] = arr[0] + arr[1]
        dp[2] = max(arr[2]+arr[0], arr[2]+arr[1], dp[1])

        for i in range(3,n):
            dp[i] = max(dp[i-2]+arr[i], dp[i-3]+arr[i]+arr[i-1], dp[i-1] )
    
        print(dp[n-1])
        
n= int(input().strip())
arr = [ int(input()) for _ in range(n) ] # [0,1,2]

if __name__ == "__main__":
    solution(n,arr)
