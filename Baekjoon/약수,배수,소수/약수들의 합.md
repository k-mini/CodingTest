import sys
input = sys.stdin.readline
from collections import defaultdict
a_dict = defaultdict(list)

for i in range(1,1001):
    for j in range(i+1, (100000//i)+1 ):
        a_dict[i*j].append(i)
        a_dict[i*j].append(j)
    a_dict[i*i].append(i)        

while True:
    n = int(input().rstrip())
    if n == -1:
        break
    
    a_dict[n].sort()
    
    if sum(a_dict[n][:-1]) == n:
        print(str(n) + " = " + ' + '.join(map(str,a_dict[n][:-1])) )
    else:
        print(str(n) + " is NOT perfect.")
